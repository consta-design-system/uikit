import { cnMixPopoverArrow } from '@consta/uikit/MixPopoverArrow';

С помощью миксина `MixPopoverArrow` можно добавить стрелочку, указывающую, к чему относится элемент.

У стрелочки можно регулировать:

- размер — в CSS-переменной `--popover-arrow-size`,
- отступ — `--popover-arrow-offset`,
- цвет — `--popover-arrow-bg-color`.

## Импорт

```tsx
import { cnMixPopoverArrow } from '@consta/uikit/MixPopoverArrow';
```

## Список свойств

```tsx
type Direction =
  | 'downCenter'
  | 'upCenter'
  | 'downRight'
  | 'downLeft'
  | 'upRight'
  | 'upLeft'
  | 'leftUp'
  | 'leftCenter'
  | 'leftDown'
  | 'rightUp'
  | 'rightCenter'
  | 'rightDown'
  | 'downStartLeft'
  | 'upStartLeft'
  | 'downStartRight'
  | 'upStartRight'
  | 'leftStartUp'’
  | 'leftStartDown'
  | 'rightStartUp'
  | 'rightStartDown';
```

| Свойство     | Тип или варианты значения | По умолчанию | Описание                                                                  |
| ------------ | ------------------------- | ------------ | ------------------------------------------------------------------------- |
| `direction?` | [`Direction`](direction)  | -            | см. [Popover](##LIBS.LIB.STAND/lib:uikit/stand:components-popover-stable) |

## Пример использования

```css
.MixPopoverArrowExample {
  --popover-bg-color: var(--color-bg-ghost);
  --popover-arrow-bg-color: var(--popover-bg-color);
  --popover-arrow-size: 10px;
  --popover-arrow-offset: 6px;
  background: var(--popover-bg-color);
}
```

```tsx
import React, { useRef } from 'react';

import { Button } from '@consta/uikit/Button';
import { Popover } from '@consta/uikit/Popover';
import { useFlag } from '@consta/uikit/useFlag';
import { Text } from '@consta/uikit/Text';

import { cnMixPopoverArrow } from '@consta/uikit/MixPopoverArrow';

export const MixPopoverArrowExample = () => {
  const [open, setOpen] = useFlag(false);
  const buttonRef = useRef<HTMLButtonElement>(null);

  return (
    <>
      <Button label="Открыть" onClick={setOpen.toggle} ref={buttonRef} />
      {open && (
        <Popover
          className={cnMixPopoverArrowExample()}
          anchorRef={buttonRef}
          direction="downCenter"
          arrowOffset={16}
          offset={14}
        >
          {(direction) => {
            return (
              <>
                <Text>Позиция - {direction}</Text>
                <div className={cnMixPopoverArrow({ direction })} />
              </>
            );
          }}
        </Popover>
      )}
    </>
  );
};
```
